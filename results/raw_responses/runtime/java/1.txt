import java.util.*;
import java.lang.*;

class Solution {
    public List<String> separateParenGroups(String paren_string) {
        int n = paren_string.length();
        List<String> result = new ArrayList<>();
        StringBuilder current = new StringBuilder(n);
        int depth = 0;

        for (int i = 0; i < n; i++) {
            char c = paren_string.charAt(i);
            if (c == '(') {
                depth++;
                current.append(c);
            } else if (c == ')') {
                current.append(c);
                depth--;
                if (depth == 0) {
                    result.add(current.toString());
                    current.setLength(0);
                }
            }
        }

        return result;
    }
}
