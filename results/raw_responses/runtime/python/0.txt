from typing import List


def has_close_elements(numbers: List[float], threshold: float) -> bool:
    if len(numbers) < 2 or not threshold > 0:
        return False

    nums = [x for x in numbers if x == x]
    if len(nums) < 2:
        return False

    nums.sort()
    prev = nums[0]
    for i in range(1, len(nums)):
        curr = nums[i]
        if curr - prev < threshold:
            return True
        prev = curr

    return False
