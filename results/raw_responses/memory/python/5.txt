from typing import List


def intersperse(numbers: List[int], delimeter: int) -> List[int]:
    n = len(numbers)
    if n == 0:
        return []

    result = [0] * (2 * n - 1)
    result[0] = numbers[0]

    j = 1
    for i in range(1, n):
        result[j] = delimeter
        result[j + 1] = numbers[i]
        j += 2

    return result
