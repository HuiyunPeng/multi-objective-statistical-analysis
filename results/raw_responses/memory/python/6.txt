from typing import List


def parse_nested_parens(paren_string: str) -> List[int]:
    result = []
    depth = 0
    max_depth = 0
    in_group = False

    for c in paren_string:
        if c == ' ':
            if in_group:
                result.append(max_depth)
                depth = 0
                max_depth = 0
                in_group = False
        elif c == '(':
            in_group = True
            depth += 1
            if depth > max_depth:
                max_depth = depth
        else:
            in_group = True
            depth -= 1

    if in_group:
        result.append(max_depth)

    return result
