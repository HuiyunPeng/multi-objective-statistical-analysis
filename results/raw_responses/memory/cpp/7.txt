#include <vector>
#include <string>
#include <utility>

using namespace std;

vector<string> filter_by_substring(vector<string> strings, string substring) {
    if (substring.empty()) return strings;

    vector<string> out;
    for (string& s : strings) {
        if (s.find(substring) != string::npos) {
            out.emplace_back(std::move(s));
        }
    }
    return out;
}
