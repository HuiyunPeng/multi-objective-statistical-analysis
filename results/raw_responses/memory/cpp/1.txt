#include <vector>
#include <string>
using namespace std;

vector<string> separate_paren_groups(string paren_string) {
    vector<string> all_parens;
    int level = 0;

    for (char c : paren_string) {
        if (c == '(') {
            if (level == 0) all_parens.emplace_back();
            ++level;
            all_parens.back().push_back(c);
        } else if (c == ')') {
            all_parens.back().push_back(c);
            --level;
        }
    }

    return all_parens;
}
