#include <vector>
#include <algorithm>
using namespace std;

bool has_close_elements(vector<float> numbers, float threshold) {
    if (threshold <= 0.0f || numbers.size() < 2) return false;

    sort(numbers.begin(), numbers.end());
    for (size_t i = 1, n = numbers.size(); i < n; ++i) {
        if (numbers[i] - numbers[i - 1] < threshold) return true;
    }
    return false;
}
