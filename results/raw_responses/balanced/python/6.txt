from typing import List


def parse_nested_parens(paren_string: str) -> List[int]:
    res = []
    depth = 0
    max_depth = 0
    in_group = False

    for c in paren_string:
        if c == '(':
            depth += 1
            in_group = True
            if depth > max_depth:
                max_depth = depth
        elif c == ')':
            depth -= 1
            in_group = True
        elif c == ' ':
            if in_group:
                res.append(max_depth)
                depth = 0
                max_depth = 0
                in_group = False
        else:
            depth -= 1
            in_group = True

    if in_group:
        res.append(max_depth)

    return res
