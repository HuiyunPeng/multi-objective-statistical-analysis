from typing import List


def separate_paren_groups(paren_string: str) -> List[str]:
    result = []
    buf = []
    depth = 0
    res_append = result.append
    buf_append = buf.append

    for c in paren_string:
        if c == '(':
            depth += 1
            buf_append(c)
        elif c == ')':
            depth -= 1
            buf_append(c)
            if depth == 0:
                res_append(''.join(buf))
                buf = []
                buf_append = buf.append

    return result
